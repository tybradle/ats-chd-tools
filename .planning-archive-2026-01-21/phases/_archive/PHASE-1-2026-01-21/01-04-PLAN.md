### Step 3: BOM Module Page with Scope Check

**File**: `src/pages/BomModule.tsx`

```typescript
import { useEffect } from 'react';
import { useActiveScopeStore } from '@/stores/active-scope-store';
import { ProjectManagerDialog } from '@/components/ProjectManagerDialog';
import { BomWorkspace } from '@/components/BomWorkspace';
import { useSearchParams } from 'react-router';

export function BomModule() {
  const { hasScope } = useActiveScopeStore();
  const [searchParams] = useSearchParams();
  const shouldManageScope = searchParams.get('manage-scope') === 'true';
  const showProjectManager = !hasScope() || shouldManageScope;

  return (
    <div className="min-h-screen bg-gray-50">
      {/* Header with scope info */}
      <header className="bg-white border-b">
        <div className="max-w-7xl mx-auto px-4 sm:px-6 lg:px-8 py-4">
          <div className="flex items-center justify-between">
            <div>
              <h1 className="text-2xl font-bold text-gray-900">BOM Translation</h1>
              <ScopeInfo />
            </div>
          </div>
        </div>
      </header>

      {/* Main content */}
      <main className="max-w-7xl mx-auto px-4 sm:px-6 lg:px-8 py-6">
        {showProjectManager ? (
          <div className="flex items-center justify-center min-h-[600px]">
            <ProjectManagerDialog
              open={true}
              onClose={() => {
                // Cannot close without selecting scope
              }}
              blocking={true}
            />
          </div>
        ) : (
          <BomWorkspace />
        )}
      </main>
    </div>
  );
}

function ScopeInfo() {
  const { projectName, packageName } = useActiveScopeStore();

  if (!hasScope()) {
    return (
      <p className="text-sm text-gray-600 mt-1">No project/package selected</p>
    );
  }

  return (
    <p className="text-sm text-gray-600 mt-1">
      {projectName} / {packageName}
    </p>
  );
}
```

### Step 4: Project Manager Dialog

**File**: `src/components/ProjectManagerDialog.tsx`

```typescript
import { useState } from 'react';
import { X } from 'lucide-react';
import { useProjectStore } from '@/stores/project-store';
import { usePackageStore } from '@/stores/package-store';
import { useActiveScopeStore } from '@/stores/active-scope-store';
import { Button } from '@/components/ui/button';
import { Dialog, DialogContent } from '@/components/ui/dialog';

interface ProjectManagerDialogProps {
  open: boolean;
  onClose: () => void;
  blocking?: boolean;
}

export function ProjectManagerDialog({
  open,
  onClose,
  blocking = false,
}: ProjectManagerDialogProps) {
  const [tab, setTab] = useState<'projects' | 'packages'>('projects');
  const { projects, fetchProjects, createProject, updateProject, deleteProject } =
    useProjectStore();
  const { packages, fetchPackages, createPackage, updatePackage, deletePackage } =
    usePackageStore();
  const { projectId, setProject, setPackage, hasScope } = useActiveScopeStore();

  const [selectedProjectId, setSelectedProjectId] = useState<number | null>(
    projectId
  );

  // Fetch projects on mount
  useEffect(() => {
    fetchProjects();
  }, [fetchProjects]);

  // Fetch packages when project selected
  useEffect(() => {
    if (selectedProjectId) {
      fetchPackages(selectedProjectId);
    }
  }, [selectedProjectId, fetchPackages]);

  const handleSelectProject = (id: number, name: string) => {
    setSelectedProjectId(id);
    setTab('packages');
  };

  const handleSelectPackage = (id: number, name: string) => {
    if (selectedProjectId) {
      const project = projects.find((p) => p.id === selectedProjectId);
      if (project) {
        setProject(selectedProjectId, project.job_number);
        setPackage(id, name);
        if (!blocking) {
          onClose();
        }
      }
    }
  };

  const canClose = !blocking || hasScope();

  return (
    <Dialog open={open} onOpenChange={canClose ? onClose : undefined}>
      <DialogContent className="max-w-4xl max-h-[80vh] overflow-hidden">
        {/* Header */}
        <div className="flex items-center justify-between border-b pb-4">
          <div>
            <h2 className="text-xl font-semibold">Project Manager</h2>
            <p className="text-sm text-gray-600 mt-1">
              {blocking
                ? 'Select a Project and Package to continue'
                : 'Manage your projects and packages'}
            </p>
          </div>
          {!blocking && (
            <button
              onClick={onClose}
              className="text-gray-400 hover:text-gray-600"
            >
              <X className="w-5 h-5" />
            </button>
          )}
        </div>

        {/* Tabs */}
        <div className="flex border-b">
          <button
            onClick={() => setTab('projects')}
            className={`
              px-4 py-2 font-medium text-sm
              ${tab === 'projects'
                ? 'border-b-2 border-blue-600 text-blue-600'
                : 'text-gray-600 hover:text-gray-900'
              }
            `}
          >
            Projects
          </button>
          <button
            onClick={() => setTab('packages')}
            disabled={!selectedProjectId}
            className={`
              px-4 py-2 font-medium text-sm
              ${tab === 'packages'
                ? 'border-b-2 border-blue-600 text-blue-600'
                : 'text-gray-600 hover:text-gray-900'
              }
              disabled:opacity-50 disabled:cursor-not-allowed
            `}
          >
            Packages
          </button>
        </div>

        {/* Content */}
        <div className="flex-1 overflow-y-auto py-4">
          {tab === 'projects' && (
            <ProjectList
              projects={projects}
              onSelect={handleSelectProject}
              onCreate={createProject}
              onUpdate={updateProject}
              onDelete={deleteProject}
              selectedId={selectedProjectId}
            />
          )}

          {tab === 'packages' && selectedProjectId && (
            <PackageList
              projectId={selectedProjectId}
              packages={packages}
              onSelect={handleSelectPackage}
              onCreate={createPackage}
              onUpdate={updatePackage}
              onDelete={deletePackage}
            />
          )}
        </div>

        {/* Footer */}
        {blocking && hasScope() && (
          <div className="flex justify-end border-t pt-4">
            <Button onClick={onClose}>Continue</Button>
          </div>
        )}
      </DialogContent>
    </Dialog>
  );
}

```
